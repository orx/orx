75 static orxINLINE orxOBOX * orxOBox_2DSet(orxOBOX *_pstRes, const orxVECTOR *_pvWorldPosition, const orxVECTOR *_pvPivot, const orxVECTOR *_pvSize, orxFLOAT _fAngle)
76 {
77  orxFLOAT fCos, fSin;
78 
79  /* Checks */
80  orxASSERT(_pstRes != orxNULL);
81  orxASSERT(_pvWorldPosition != orxNULL);
82  orxASSERT(_pvPivot != orxNULL);
83 
84  /* Gets cosine and sine */
85  if(_fAngle == orxFLOAT_0)
86  {
87  fCos = orxFLOAT_1;
88  fSin = orxFLOAT_0;
89  }
90  else
91  {
92  fCos = orxMath_Cos(_fAngle);
93  fSin = orxMath_Sin(_fAngle);
94  }
95 
96  /* Sets axis */
97  orxVector_Set(&(_pstRes->vX), fCos * _pvSize->fX, fSin * _pvSize->fX, orxFLOAT_0);
98  orxVector_Set(&(_pstRes->vY), -fSin * _pvSize->fY, fCos * _pvSize->fY, orxFLOAT_0);
99  orxVector_Set(&(_pstRes->vZ), orxFLOAT_0, orxFLOAT_0, _pvSize->fZ);
100 
101  /* Sets pivot */
102  orxVector_Set(&(_pstRes->vPivot), (fCos * _pvPivot->fX) - (fSin * _pvPivot->fY), (fSin * _pvPivot->fX) + (fCos * _pvPivot->fY), _pvPivot->fZ);
103 
104  /* Sets box position */
105  orxVector_Copy(&(_pstRes->vPosition), _pvWorldPosition);
106 
107  /* Done! */
108  return _pstRes;
109 }

286 static orxINLINE orxBOOL orxOBox_ZAlignedTestIntersection(const orxOBOX *_pstBox1, const orxOBOX *_pstBox2)
287 {
288  register orxBOOL bResult;
289 
290  /* Checks */
291  orxASSERT(_pstBox1 != orxNULL);
292  orxASSERT(_pstBox2 != orxNULL);
293  orxASSERT(_pstBox1->vZ.fX == orxFLOAT_0);
294  orxASSERT(_pstBox1->vZ.fY == orxFLOAT_0);
295  orxASSERT(_pstBox1->vZ.fZ >= orxFLOAT_0);
296 
297  /* Z intersected? */
298  if((_pstBox2->vPosition.fZ + _pstBox2->vZ.fZ >= _pstBox1->vPosition.fZ)
299  && (_pstBox2->vPosition.fZ <= _pstBox1->vPosition.fZ + _pstBox1->vZ.fZ))
300  {
301  orxU32 i;
302  orxVECTOR vOrigin1, vOrigin2, *pvOrigin1 = &vOrigin1, *pvOrigin2 = &vOrigin2, *pvTemp;
303  const orxOBOX *pstBox1 = _pstBox1, *pstBox2 = _pstBox2, *pstTemp;
304 
305  /* Computes boxes origins */
306  vOrigin1.fX = _pstBox1->vPosition.fX - pstBox1->vPivot.fX;
307  vOrigin1.fY = _pstBox1->vPosition.fY - pstBox1->vPivot.fY;
308  vOrigin2.fX = _pstBox2->vPosition.fX - pstBox2->vPivot.fX;
309  vOrigin2.fY = _pstBox2->vPosition.fY - pstBox2->vPivot.fY;
310 
311  /* Test each box against the other */
312  for(i = 2, bResult = orxTRUE;
313  i != 0;
314  i--, pstTemp = pstBox1, pstBox1 = pstBox2, pstBox2 = pstTemp, pvTemp = pvOrigin1, pvOrigin1 = pvOrigin2, pvOrigin2 = pvTemp)
315  {
316  orxVECTOR vToCorner[4];
317  const orxVECTOR *pvAxis;
318  orxU32 j;
319 
320  /* Gets to-corner vectors */
321  vToCorner[0].fX = pvOrigin2->fX - pvOrigin1->fX;
322  vToCorner[0].fY = pvOrigin2->fY - pvOrigin1->fY;
323  vToCorner[1].fX = vToCorner[0].fX + pstBox2->vX.fX;
324  vToCorner[1].fY = vToCorner[0].fY + pstBox2->vX.fY;
325  vToCorner[2].fX = vToCorner[1].fX + pstBox2->vY.fX;
326  vToCorner[2].fY = vToCorner[1].fY + pstBox2->vY.fY;
327  vToCorner[3].fX = vToCorner[0].fX + pstBox2->vY.fX;
328  vToCorner[3].fY = vToCorner[0].fY + pstBox2->vY.fY;
329 
330  /* For both axis */
331  for(j = 2, pvAxis = &(pstBox1->vX);
332  j != 0;
333  j--, pvAxis++)
334  {
335  orxFLOAT fMin, fMax, fProj;
336  orxU32 k;
337 
338  /* Gets initial projected values */
339  fMin = fMax = fProj = orxVector_2DDot(&vToCorner[0], pvAxis);
340 
341  /* For all remaining corners */
342  for(k = 1; k < 4; k++)
343  {
344  /* Gets projected value */
345  fProj = orxVector_2DDot(&vToCorner[k], pvAxis);
346 
347  /* Updates extrema */
348  if(fProj > fMax)
349  {
350  fMax = fProj;
351  }
352  else if(fProj < fMin)
353  {
354  fMin = fProj;
355  }
356  }
357 
358  /* Not intersecting? */
359  if((fMax < orxFLOAT_0)
360  || (fMin > orxVector_GetSquareSize(pvAxis)))
361  {
362  /* Updates result */
363  bResult = orxFALSE;
364  break;
365  }
366  }
367  }
368  }
369  else
370  {
371  /* Updates result */
372  bResult = orxFALSE;
373  }
374 
375  /* Done! */
376  return bResult;
377 }
