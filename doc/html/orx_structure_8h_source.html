398 static orxINLINE void orxStructure_IncreaseCounter(void *_pStructure)
399 {
400  orxU64 u64Counter;
401 
402  /* Checks */
403  orxSTRUCTURE_ASSERT(_pStructure);
404 
405  /* Gets current counter */
406  u64Counter = (orxSTRUCTURE(_pStructure)->u64GUID & orxSTRUCTURE_GUID_MASK_REF_COUNTER) >> orxSTRUCTURE_GUID_SHIFT_REF_COUNTER;
407 
408  /* Updates it */
409  u64Counter++;
410 
411  /* Checks */
412  orxASSERT(u64Counter <= (orxSTRUCTURE_GUID_MASK_REF_COUNTER >> orxSTRUCTURE_GUID_SHIFT_REF_COUNTER));
413 
414  /* Stores it */
415  orxSTRUCTURE(_pStructure)->u64GUID = (orxSTRUCTURE(_pStructure)->u64GUID & ~orxSTRUCTURE_GUID_MASK_REF_COUNTER) | (u64Counter << orxSTRUCTURE_GUID_SHIFT_REF_COUNTER);
416 
417  /* Done! */
418  return;
419 }
