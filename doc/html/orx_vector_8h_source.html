573 static orxINLINE orxVECTOR * orxVector_2DRotate(orxVECTOR *_pvRes, const orxVECTOR *_pvOp, orxFLOAT _fAngle)
574 {
575  /* Checks */
576  orxASSERT(_pvRes != orxNULL);
577  orxASSERT(_pvOp != orxNULL);
578 
579  /* 0? */
580  if(_fAngle == orxFLOAT_0)
581  {
582  /* Updates result */
583  orxVector_Copy(_pvRes, _pvOp);
584  }
585  /* PI/2? */
586  if(_fAngle == orxMATH_KF_PI_BY_2)
587  {
588  /* Updates result */
589  orxVector_Set(_pvRes, -_pvOp->fY, _pvOp->fX, _pvOp->fZ);
590  }
591  /* -PI/2? */
592  else if(_fAngle == -orxMATH_KF_PI_BY_2)
593  {
594  /* Updates result */
595  orxVector_Set(_pvRes, _pvOp->fY, -_pvOp->fX, _pvOp->fZ);
596  }
597  else if(_fAngle == orxMATH_KF_PI)
598  {
599  /* Updates result */
600  orxVector_Set(_pvRes, -_pvOp->fX, -_pvOp->fY, _pvOp->fZ);
601  }
602  /* Any other angle */
603  else
604  {
605  orxFLOAT fSin, fCos;
606 
607  /* Gets cos & sin of angle */
608  fCos = orxMath_Cos(_fAngle);
609  fSin = orxMath_Sin(_fAngle);
610 
611  /* Updates result */
612  orxVector_Set(_pvRes, (fCos * _pvOp->fX) - (fSin * _pvOp->fY), (fSin * _pvOp->fX) + (fCos * _pvOp->fY), _pvOp->fZ);
613  }
614 
615  /* Done! */
616  return _pvRes;
617 }

662 static orxINLINE orxVECTOR * orxVector_FromCartesianToSpherical(orxVECTOR *_pvRes, const orxVECTOR *_pvOp)
663 {
664  /* Checks */
665  orxASSERT(_pvRes != orxNULL);
666  orxASSERT(_pvOp != orxNULL);
667 
668  /* Is operand vector null? */
669  if((_pvOp->fX == orxFLOAT_0)
670  && (_pvOp->fY == orxFLOAT_0)
671  && (_pvOp->fZ == orxFLOAT_0))
672  {
673  /* Updates result vector */
674  _pvRes->fRho = _pvRes->fTheta = _pvRes->fPhi = orxFLOAT_0;
675  }
676  else
677  {
678  orxFLOAT fRho, fTheta, fPhi;
679 
680  /* Z = 0? */
681  if(_pvOp->fZ == orxFLOAT_0)
682  {
683  /* X = 0? */
684  if(_pvOp->fX == orxFLOAT_0)
685  {
686  /* Gets absolute value */
687  fRho = orxMath_Abs(_pvOp->fY);
688  }
689  /* X != 0 and Y = 0? */
690  else if(_pvOp->fY == orxFLOAT_0)
691  {
692  /* Gets absolute value */
693  fRho = orxMath_Abs(_pvOp->fX);
694  }
695  /* X != 0 and Y != 0 */
696  else
697  {
698  /* Computes rho */
699  fRho = orxMath_Sqrt((_pvOp->fX * _pvOp->fX) + (_pvOp->fY * _pvOp->fY));
700  }
701 
702  /* Sets phi */
703  fPhi = orxFLOAT_0;
704  }
705  else
706  {
707  /* X = 0 and Y = 0? */
708  if((_pvOp->fX == orxFLOAT_0) && (_pvOp->fY == orxFLOAT_0))
709  {
710  /* Z < 0? */
711  if(_pvOp->fZ < orxFLOAT_0)
712  {
713  /* Gets absolute value */
714  fRho = orxMath_Abs(_pvOp->fZ);
715 
716  /* Sets phi */
717  fPhi = orxMATH_KF_PI_BY_2;
718  }
719  else
720  {
721  /* Sets rho */
722  fRho = _pvOp->fZ;
723 
724  /* Sets phi */
725  fPhi = -orxMATH_KF_PI_BY_2;
726  }
727  }
728  else
729  {
730  /* Computes rho */
731  fRho = orxMath_Sqrt(orxVector_GetSquareSize(_pvOp));
732 
733  /* Computes phi */
734  fPhi = orxMath_ACos(_pvOp->fZ / fRho) - orxMATH_KF_PI_BY_2;
735  }
736  }
737 
738  /* Computes theta */
739  fTheta = orxMath_ATan(_pvOp->fY, _pvOp->fX);
740 
741  /* Updates result */
742  _pvRes->fRho = fRho;
743  _pvRes->fTheta = fTheta;
744  _pvRes->fPhi = fPhi;
745  }
746 
747  /* Done! */
748  return _pvRes;
749 }

756 static orxINLINE orxVECTOR * orxVector_FromSphericalToCartesian(orxVECTOR *_pvRes, const orxVECTOR *_pvOp)
757 {
758  orxFLOAT fSinPhi, fCosPhi, fSinTheta, fCosTheta, fRho;
759 
760  /* Checks */
761  orxASSERT(_pvRes != orxNULL);
762  orxASSERT(_pvOp != orxNULL);
763 
764  /* Stores rho */
765  fRho = _pvOp->fRho;
766 
767  /* Gets sine & cosine */
768  fSinTheta = orxMath_Sin(_pvOp->fTheta);
769  fCosTheta = orxMath_Cos(_pvOp->fTheta);
770  fSinPhi = orxMath_Sin(_pvOp->fPhi + orxMATH_KF_PI_BY_2);
771  fCosPhi = orxMath_Cos(_pvOp->fPhi + orxMATH_KF_PI_BY_2);
772  if(orxMath_Abs(fSinTheta) < orxMATH_KF_EPSILON)
773  {
774  fSinTheta = orxFLOAT_0;
775  }
776  if(orxMath_Abs(fCosTheta) < orxMATH_KF_EPSILON)
777  {
778  fCosTheta = orxFLOAT_0;
779  }
780  if(orxMath_Abs(fSinPhi) < orxMATH_KF_EPSILON)
781  {
782  fSinPhi = orxFLOAT_0;
783  }
784  if(orxMath_Abs(fCosPhi) < orxMATH_KF_EPSILON)
785  {
786  fCosPhi = orxFLOAT_0;
787  }
788 
789  /* Updates result */
790  _pvRes->fX = fRho * fCosTheta * fSinPhi;
791  _pvRes->fY = fRho * fSinTheta * fSinPhi;
792  _pvRes->fZ = fRho * fCosPhi;
793 
794  /* Done! */
795  return _pvRes;
796 }
