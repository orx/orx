573 static orxINLINE orxVECTOR * orxVector_2DRotate(orxVECTOR *_pvRes, const orxVECTOR *_pvOp, orxFLOAT _fAngle)
574 {
575  /* Checks */
576  orxASSERT(_pvRes != orxNULL);
577  orxASSERT(_pvOp != orxNULL);
578 
579  /* PI/2? */
580  if(_fAngle == orxMATH_KF_PI_BY_2)
581  {
582  /* Updates result */
583  orxVector_Set(_pvRes, -_pvOp->fY, _pvOp->fX, _pvOp->fZ);
584  }
585  /* -PI/2? */
586  else if(_fAngle == -orxMATH_KF_PI_BY_2)
587  {
588  /* Updates result */
589  orxVector_Set(_pvRes, _pvOp->fY, -_pvOp->fX, _pvOp->fZ);
590  }
591  /* Any other angle */
592  else
593  {
594  orxFLOAT fSin, fCos;
595 
596  /* Gets cos & sin of angle */
597  fCos = orxMath_Cos(_fAngle);
598  fSin = orxMath_Sin(_fAngle);
599 
600  /* Updates result */
601  orxVector_Set(_pvRes, (fCos * _pvOp->fX) - (fSin * _pvOp->fY), (fSin * _pvOp->fX) + (fCos * _pvOp->fY), _pvOp->fZ);
602  }
603 
604  /* Done! */
605  return _pvRes;
606 }

651 static orxINLINE orxVECTOR * orxVector_FromCartesianToSpherical(orxVECTOR *_pvRes, const orxVECTOR *_pvOp)
652 {
653  /* Checks */
654  orxASSERT(_pvRes != orxNULL);
655  orxASSERT(_pvOp != orxNULL);
656 
657  /* Is operand vector null? */
658  if((_pvOp->fX == orxFLOAT_0)
659  && (_pvOp->fY == orxFLOAT_0)
660  && (_pvOp->fZ == orxFLOAT_0))
661  {
662  /* Updates result vector */
663  _pvRes->fRho = _pvRes->fTheta = _pvRes->fPhi = orxFLOAT_0;
664  }
665  else
666  {
667  orxFLOAT fRho, fTheta, fPhi;
668 
669  /* Z = 0? */
670  if(_pvOp->fZ == orxFLOAT_0)
671  {
672  /* X = 0? */
673  if(_pvOp->fX == orxFLOAT_0)
674  {
675  /* Gets absolute value */
676  fRho = orxMath_Abs(_pvOp->fY);
677  }
678  /* X != 0 and Y = 0? */
679  else if(_pvOp->fY == orxFLOAT_0)
680  {
681  /* Gets absolute value */
682  fRho = orxMath_Abs(_pvOp->fX);
683  }
684  /* X != 0 and Y != 0 */
685  else
686  {
687  /* Computes rho */
688  fRho = orxMath_Sqrt((_pvOp->fX * _pvOp->fX) + (_pvOp->fY * _pvOp->fY));
689  }
690 
691  /* Sets phi */
692  fPhi = orxFLOAT_0;
693  }
694  else
695  {
696  /* X = 0 and Y = 0? */
697  if((_pvOp->fX == orxFLOAT_0) && (_pvOp->fY == orxFLOAT_0))
698  {
699  /* Z < 0? */
700  if(_pvOp->fZ < orxFLOAT_0)
701  {
702  /* Gets absolute value */
703  fRho = orxMath_Abs(_pvOp->fZ);
704 
705  /* Sets phi */
706  fPhi = orxMATH_KF_PI_BY_2;
707  }
708  else
709  {
710  /* Sets rho */
711  fRho = _pvOp->fZ;
712 
713  /* Sets phi */
714  fPhi = -orxMATH_KF_PI_BY_2;
715  }
716  }
717  else
718  {
719  /* Computes rho */
720  fRho = orxMath_Sqrt(orxVector_GetSquareSize(_pvOp));
721 
722  /* Computes phi */
723  fPhi = orxMath_ACos(_pvOp->fZ / fRho) - orxMATH_KF_PI_BY_2;
724  }
725  }
726 
727  /* Computes theta */
728  fTheta = orxMath_ATan(_pvOp->fY, _pvOp->fX);
729 
730  /* Updates result */
731  _pvRes->fRho = fRho;
732  _pvRes->fTheta = fTheta;
733  _pvRes->fPhi = fPhi;
734  }
735 
736  /* Done! */
737  return _pvRes;
738 }

745 static orxINLINE orxVECTOR * orxVector_FromSphericalToCartesian(orxVECTOR *_pvRes, const orxVECTOR *_pvOp)
746 {
747  orxFLOAT fSinPhi, fCosPhi, fSinTheta, fCosTheta, fRho;
748 
749  /* Checks */
750  orxASSERT(_pvRes != orxNULL);
751  orxASSERT(_pvOp != orxNULL);
752 
753  /* Stores rho */
754  fRho = _pvOp->fRho;
755 
756  /* Gets sine & cosine */
757  fSinTheta = orxMath_Sin(_pvOp->fTheta);
758  fCosTheta = orxMath_Cos(_pvOp->fTheta);
759  fSinPhi = orxMath_Sin(_pvOp->fPhi + orxMATH_KF_PI_BY_2);
760  fCosPhi = orxMath_Cos(_pvOp->fPhi + orxMATH_KF_PI_BY_2);
761  if(orxMath_Abs(fSinTheta) < orxMATH_KF_EPSILON)
762  {
763  fSinTheta = orxFLOAT_0;
764  }
765  if(orxMath_Abs(fCosTheta) < orxMATH_KF_EPSILON)
766  {
767  fCosTheta = orxFLOAT_0;
768  }
769  if(orxMath_Abs(fSinPhi) < orxMATH_KF_EPSILON)
770  {
771  fSinPhi = orxFLOAT_0;
772  }
773  if(orxMath_Abs(fCosPhi) < orxMATH_KF_EPSILON)
774  {
775  fCosPhi = orxFLOAT_0;
776  }
777 
778  /* Updates result */
779  _pvRes->fX = fRho * fCosTheta * fSinPhi;
780  _pvRes->fY = fRho * fSinTheta * fSinPhi;
781  _pvRes->fZ = fRho * fCosPhi;
782 
783  /* Done! */
784  return _pvRes;
785 }
