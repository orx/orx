643 static orxINLINE orxVECTOR * orxVector_2DRotate(orxVECTOR *_pvRes, const orxVECTOR *_pvOp, orxFLOAT _fAngle)
644 {
645  /* Checks */
646  orxASSERT(_pvRes != orxNULL);
647  orxASSERT(_pvOp != orxNULL);
648 
649  /* 0? */
650  if(_fAngle == orxFLOAT_0)
651  {
652  /* Updates result */
653  orxVector_Copy(_pvRes, _pvOp);
654  }
655  /* PI/2? */
656  if(_fAngle == orxMATH_KF_PI_BY_2)
657  {
658  /* Updates result */
659  orxVector_Set(_pvRes, -_pvOp->fY, _pvOp->fX, _pvOp->fZ);
660  }
661  /* -PI/2? */
662  else if(_fAngle == -orxMATH_KF_PI_BY_2)
663  {
664  /* Updates result */
665  orxVector_Set(_pvRes, _pvOp->fY, -_pvOp->fX, _pvOp->fZ);
666  }
667  else if(_fAngle == orxMATH_KF_PI)
668  {
669  /* Updates result */
670  orxVector_Set(_pvRes, -_pvOp->fX, -_pvOp->fY, _pvOp->fZ);
671  }
672  /* Any other angle */
673  else
674  {
675  orxFLOAT fSin, fCos;
676 
677  /* Gets cos & sin of angle */
678  fCos = orxMath_Cos(_fAngle);
679  fSin = orxMath_Sin(_fAngle);
680 
681  /* Updates result */
682  orxVector_Set(_pvRes, (fCos * _pvOp->fX) - (fSin * _pvOp->fY), (fSin * _pvOp->fX) + (fCos * _pvOp->fY), _pvOp->fZ);
683  }
684 
685  /* Done! */
686  return _pvRes;
687 }

732 static orxINLINE orxVECTOR * orxVector_FromCartesianToSpherical(orxVECTOR *_pvRes, const orxVECTOR *_pvOp)
733 {
734  /* Checks */
735  orxASSERT(_pvRes != orxNULL);
736  orxASSERT(_pvOp != orxNULL);
737 
738  /* Is operand vector null? */
739  if((_pvOp->fX == orxFLOAT_0)
740  && (_pvOp->fY == orxFLOAT_0)
741  && (_pvOp->fZ == orxFLOAT_0))
742  {
743  /* Updates result vector */
744  _pvRes->fRho = _pvRes->fTheta = _pvRes->fPhi = orxFLOAT_0;
745  }
746  else
747  {
748  orxFLOAT fRho, fTheta, fPhi;
749 
750  /* Z = 0? */
751  if(_pvOp->fZ == orxFLOAT_0)
752  {
753  /* X = 0? */
754  if(_pvOp->fX == orxFLOAT_0)
755  {
756  /* Gets absolute value */
757  fRho = orxMath_Abs(_pvOp->fY);
758  }
759  /* X != 0 and Y = 0? */
760  else if(_pvOp->fY == orxFLOAT_0)
761  {
762  /* Gets absolute value */
763  fRho = orxMath_Abs(_pvOp->fX);
764  }
765  /* X != 0 and Y != 0 */
766  else
767  {
768  /* Computes rho */
769  fRho = orxMath_Sqrt((_pvOp->fX * _pvOp->fX) + (_pvOp->fY * _pvOp->fY));
770  }
771 
772  /* Sets phi */
773  fPhi = orxFLOAT_0;
774  }
775  else
776  {
777  /* X = 0 and Y = 0? */
778  if((_pvOp->fX == orxFLOAT_0) && (_pvOp->fY == orxFLOAT_0))
779  {
780  /* Z < 0? */
781  if(_pvOp->fZ < orxFLOAT_0)
782  {
783  /* Gets absolute value */
784  fRho = orxMath_Abs(_pvOp->fZ);
785 
786  /* Sets phi */
787  fPhi = orxMATH_KF_PI_BY_2;
788  }
789  else
790  {
791  /* Sets rho */
792  fRho = _pvOp->fZ;
793 
794  /* Sets phi */
795  fPhi = -orxMATH_KF_PI_BY_2;
796  }
797  }
798  else
799  {
800  /* Computes rho */
801  fRho = orxMath_Sqrt(orxVector_GetSquareSize(_pvOp));
802 
803  /* Computes phi */
804  fPhi = orxMath_ACos(_pvOp->fZ / fRho) - orxMATH_KF_PI_BY_2;
805  }
806  }
807 
808  /* Computes theta */
809  fTheta = orxMath_ATan(_pvOp->fY, _pvOp->fX);
