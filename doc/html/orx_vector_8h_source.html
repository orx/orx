656static orxINLINE orxVECTOR * orxVector_Normalize(orxVECTOR *_pvRes, const orxVECTOR *_pvOp)
657{
658 orxFLOAT fOp;
659
660 /* Checks */
661 orxASSERT(_pvRes != orxNULL);
662 orxASSERT(_pvOp != orxNULL);
663
664 /* Gets squared size */
665 fOp = (_pvOp->fX * _pvOp->fX) + (_pvOp->fY * _pvOp->fY) + (_pvOp->fZ * _pvOp->fZ);
666
667 /* Gets reciprocal size */
668 fOp = orxFLOAT_1 / (orxMATH_KF_TINY_EPSILON + orxMath_Sqrt(fOp));
669
670 /* Updates result */
671 _pvRes->fX = fOp * _pvOp->fX;
672 _pvRes->fY = fOp * _pvOp->fY;
673 _pvRes->fZ = fOp * _pvOp->fZ;
674
675 /* Done! */
676 return _pvRes;

685static orxINLINE orxVECTOR * orxVector_2DRotate(orxVECTOR *_pvRes, const orxVECTOR *_pvOp, orxFLOAT _fAngle)
686{
687 /* Checks */
688 orxASSERT(_pvRes != orxNULL);
689 orxASSERT(_pvOp != orxNULL);
690
691 /* 0? */
692 if(_fAngle == orxFLOAT_0)
693 {
694 /* Updates result */
695 orxVector_Copy(_pvRes, _pvOp);
696 }
697 /* PI/2? */
698 if(_fAngle == orxMATH_KF_PI_BY_2)
699 {
700 /* Updates result */
701 orxVector_Set(_pvRes, -_pvOp->fY, _pvOp->fX, _pvOp->fZ);
702 }
703 /* -PI/2? */
704 else if(_fAngle == -orxMATH_KF_PI_BY_2)
705 {
706 /* Updates result */
707 orxVector_Set(_pvRes, _pvOp->fY, -_pvOp->fX, _pvOp->fZ);
708 }
709 else if(_fAngle == orxMATH_KF_PI)
710 {
711 /* Updates result */
712 orxVector_Set(_pvRes, -_pvOp->fX, -_pvOp->fY, _pvOp->fZ);
713 }
714 /* Any other angle */
715 else
716 {
717 orxFLOAT fSin, fCos;
718
719 /* Gets cos & sin of angle */
720 fCos = orxMath_Cos(_fAngle);
721 fSin = orxMath_Sin(_fAngle);
722
723 /* Updates result */
724 orxVector_Set(_pvRes, (fCos * _pvOp->fX) - (fSin * _pvOp->fY), (fSin * _pvOp->fX) + (fCos * _pvOp->fY), _pvOp->fZ);
725 }
726
727 /* Done! */
728 return _pvRes;
729}

774static orxINLINE orxVECTOR * orxVector_FromCartesianToSpherical(orxVECTOR *_pvRes, const orxVECTOR *_pvOp)
775{
776 /* Checks */
777 orxASSERT(_pvRes != orxNULL);
778 orxASSERT(_pvOp != orxNULL);
779
780 /* Is operand vector null? */
781 if((_pvOp->fX == orxFLOAT_0)
782 && (_pvOp->fY == orxFLOAT_0)
783 && (_pvOp->fZ == orxFLOAT_0))
784 {
785 /* Updates result vector */
786 _pvRes->fRho = _pvRes->fTheta = _pvRes->fPhi = orxFLOAT_0;
787 }
788 else
789 {
790 orxFLOAT fRho, fTheta, fPhi;
791
792 /* Z = 0? */
793 if(_pvOp->fZ == orxFLOAT_0)
794 {
795 /* X = 0? */
796 if(_pvOp->fX == orxFLOAT_0)
797 {
798 /* Gets absolute value */
799 fRho = orxMath_Abs(_pvOp->fY);
800 }
801 /* X != 0 and Y = 0? */
802 else if(_pvOp->fY == orxFLOAT_0)
803 {
804 /* Gets absolute value */
805 fRho = orxMath_Abs(_pvOp->fX);
806 }
807 /* X != 0 and Y != 0 */
808 else
809 {
810 /* Computes rho */
811 fRho = orxMath_Sqrt((_pvOp->fX * _pvOp->fX) + (_pvOp->fY * _pvOp->fY));
812 }
813
814 /* Sets phi */
815 fPhi = orxFLOAT_0;
816 }
817 else
818 {
819 /* X = 0 and Y = 0? */
820 if((_pvOp->fX == orxFLOAT_0) && (_pvOp->fY == orxFLOAT_0))
821 {
822 /* Z < 0? */
823 if(_pvOp->fZ < orxFLOAT_0)
824 {
825 /* Gets absolute value */
826 fRho = orxMath_Abs(_pvOp->fZ);
827
828 /* Sets phi */
829 fPhi = orxMATH_KF_PI_BY_2;
830 }
831 else
832 {
833 /* Sets rho */
834 fRho = _pvOp->fZ;
835
836 /* Sets phi */
837 fPhi = -orxMATH_KF_PI_BY_2;
838 }
839 }
840 else
841 {
842 /* Computes rho */
843 fRho = orxMath_Sqrt(orxVector_GetSquareSize(_pvOp));
844
845 /* Computes phi */
846 fPhi = orxMath_ACos(_pvOp->fZ / fRho) - orxMATH_KF_PI_BY_2;
847 }
848 }
849
850 /* Computes theta */
851 fTheta = orxMath_ATan(_pvOp->fY, _pvOp->fX);
852
853 /* Updates result */
854 _pvRes->fRho = fRho;
855 _pvRes->fTheta = fTheta;
856 _pvRes->fPhi = fPhi;
857 }
858
859 /* Done! */
860 return _pvRes;

868static orxINLINE orxVECTOR * orxVector_FromSphericalToCartesian(orxVECTOR *_pvRes, const orxVECTOR *_pvOp)
869{

871
872 /* Checks */
873 orxASSERT(_pvRes != orxNULL);
874 orxASSERT(_pvOp != orxNULL);
875

881 fCosTheta = orxMath_Cos(_pvOp->fTheta);
882 fSinPhi = orxMath_Sin(_pvOp->fPhi + orxMATH_KF_PI_BY_2);
883 fCosPhi = orxMath_Cos(_pvOp->fPhi + orxMATH_KF_PI_BY_2);
884 if(orxMath_Abs(fSinTheta) < orxMATH_KF_EPSILON)
885 {
886 fSinTheta = orxFLOAT_0;

888 if(orxMath_Abs(fCosTheta) < orxMATH_KF_EPSILON)
889 {
890 fCosTheta = orxFLOAT_0;

893 {

895 }

897 {
898 fCosPhi = orxFLOAT_0;
899 }

901 /* Updates result */
902 _pvRes->fX = fRho * fCosTheta * fSinPhi;
903 _pvRes->fY = fRho * fSinTheta * fSinPhi;
904 _pvRes->fZ = fRho * fCosPhi;
