414 static orxINLINE orxRGBA orxColor_ToRGBA(const orxCOLOR *_pstColor)
415 {
416  orxRGBA stResult;
417  orxVECTOR vColor;
418  orxFLOAT fAlpha;
419 
420  /* Checks */
421  orxASSERT(_pstColor != orxNULL);
422 
423  /* Clamps RGB components */
424  orxVector_Clamp(&vColor, &(_pstColor->vRGB), &orxVECTOR_0, &orxVECTOR_WHITE);
425 
426  /* De-normalizes vector */
427  orxVector_Mulf(&vColor, &vColor, orxCOLOR_DENORMALIZER);
428 
429  /* Clamps alpha */
430  fAlpha = orxCLAMP(_pstColor->fAlpha, orxFLOAT_0, orxFLOAT_1);
431 
432  /* Updates result */

434 
435  /* Done! */

462 static orxCOLOR *orxFASTCALL orxColor_FromRGBToHSL(orxCOLOR *_pstDst, const orxCOLOR *_pstSrc)
463 {
464  orxCOLOR *pstResult = _pstDst;
465  orxFLOAT fMin, fMax, fDelta, fR, fG, fB;
466 
467  /* Checks */
468  orxASSERT(_pstDst != orxNULL);
469  orxASSERT(_pstSrc != orxNULL);
470 
471  /* Gets source red, blue and green components */
472  fR = _pstSrc->vRGB.fR;
473  fG = _pstSrc->vRGB.fG;
474  fB = _pstSrc->vRGB.fB;
475 
476  /* Gets min, max & delta values */
477  fMin = orxMIN(fR, orxMIN(fG, fB));
478  fMax = orxMAX(fR, orxMAX(fG, fB));
479  fDelta = fMax - fMin;
480 
481  /* Stores lightness */
482  pstResult->vHSL.fL = orx2F(0.5f) * (fMax + fMin);
483 
484  /* Gray? */
485  if(fDelta == orxFLOAT_0)
486  {
487  /* Gets hue & saturation */
488  pstResult->vHSL.fH = pstResult->vHSL.fS = orxFLOAT_0;
489  }
490  else
491  {
492  /* Updates saturation */
493  pstResult->vHSL.fS = (pstResult->vHSL.fL < orx2F(0.5f))
494  ? fDelta / (fMax + fMin)
495  : fDelta / (orx2F(2.0f) - fMax - fMin);
496 
497  /* Red tone? */
498  if(fR == fMax)
499  {
500  /* Updates hue */
501  pstResult->vHSL.fH = orx2F(1.0f / 6.0f) * (fG - fB) / fDelta;
502  }
503  /* Green tone? */
504  else if(fG == fMax)
505  {
506  /* Updates hue */
507  pstResult->vHSL.fH = orx2F(1.0f / 3.0f) + (orx2F(1.0f / 6.0f) * (fB - fR) / fDelta);
508  }
509  /* Blue tone */
510  else
511  {
512  /* Updates hue */
513  pstResult->vHSL.fH = orx2F(2.0f / 3.0f) + (orx2F(1.0f / 6.0f) * (fR - fG) / fDelta);
514  }
515 
516  /* Clamps hue */
517  if(pstResult->vHSL.fH < orxFLOAT_0)
518  {
519  pstResult->vHSL.fH += orxFLOAT_1;
520  }
521  else if(pstResult->vHSL.fH > orxFLOAT_1)
522  {
523  pstResult->vHSL.fH -= orxFLOAT_1;
524  }
525  }
526 
527  /* Updates alpha */
528  pstResult->fAlpha = _pstSrc->fAlpha;
529 
530  /* Done! */
531  return pstResult;
532 }

539 static orxCOLOR *orxFASTCALL orxColor_FromHSLToRGB(orxCOLOR *_pstDst, const orxCOLOR *_pstSrc)
540 {
541  orxCOLOR *pstResult = _pstDst;
542  orxFLOAT fH, fS, fL;
543 
544 #define orxCOLOR_GET_RGB_COMPONENT(RESULT, ALT, CHROMA, HUE) \
545 do \
546 { \
547  if(HUE < orx2F(1.0f / 6.0f)) \
548  { \
549  RESULT = ALT + (orx2F(6.0f) * HUE * (CHROMA - ALT)); \
550  } \
551  else if(HUE < orx2F(1.0f / 2.0f)) \
552  { \
553  RESULT = CHROMA; \
554  } \
555  else if(HUE < orx2F(2.0f / 3.0f)) \
556  { \
557  RESULT = ALT + (orx2F(6.0f) * (CHROMA - ALT) * (orx2F(2.0f / 3.0f) - HUE)); \
558  } \
559  else \
560  { \
561  RESULT = ALT; \
562  } \
563  if(RESULT < orxMATH_KF_EPSILON) \
564  { \
565  RESULT = orxFLOAT_0; \
566  } \
567  else if(RESULT > orxFLOAT_1 - orxMATH_KF_EPSILON) \
568  { \
569  RESULT = orxFLOAT_1; \
570  } \
571 } \
572 while(orxFALSE)
573 
574  /* Checks */
575  orxASSERT(_pstDst != orxNULL);
576  orxASSERT(_pstSrc != orxNULL);
577 
578  /* Gets source hue, saturation and lightness components */
579  fH = _pstSrc->vRGB.fH;
580  fS = _pstSrc->vRGB.fS;
581  fL = _pstSrc->vRGB.fL;
582 
583  /* Gray? */
584  if(fS == orxFLOAT_0)
585  {
586  /* Updates result */
587  orxVector_SetAll(&(pstResult->vRGB), fL);
588  }
589  else
590  {
591  orxFLOAT fChroma, fIntermediate;
592 
593  /* Gets chroma */
594  fChroma = (fL < orx2F(0.5f))
595  ? fL + (fL * fS)
596  : fL + fS - (fL * fS);
597 
598  /* Gets intermediate value */
599  fIntermediate = (orx2F(2.0f) * fL) - fChroma;
600 
601  /* Gets RGB components */
602  if(fH > orx2F(2.0f / 3.0f))
603  {
604  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fR, fIntermediate, fChroma, (fH - orx2F(2.0f / 3.0f)));
605  }
606  else
607  {
608  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fR, fIntermediate, fChroma, (fH + orx2F(1.0f / 3.0f)));
609  }
610  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fG, fIntermediate, fChroma, fH);
611  if(fH < orx2F(1.0f / 3.0f))
612  {
613  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fB, fIntermediate, fChroma, (fH + orx2F(2.0f / 3.0f)));
614  }
615  else
616  {
617  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fB, fIntermediate, fChroma, (fH - orx2F(1.0f / 3.0f)));
618  }
619  }
620 
621  /* Updates alpha */
622  pstResult->fAlpha = _pstSrc->fAlpha;
623 
624  /* Done! */
625  return pstResult;
626 }

708 static orxCOLOR *orxFASTCALL orxColor_FromHSVToRGB(orxCOLOR *_pstDst, const orxCOLOR *_pstSrc)
709 {
710  orxCOLOR *pstResult = _pstDst;
711  orxFLOAT fH, fS, fV;
712 
713  /* Checks */
714  orxASSERT(_pstDst != orxNULL);
715  orxASSERT(_pstSrc != orxNULL);
716 
717  /* Gets source hue, saturation and value components */
718  fH = _pstSrc->vRGB.fH;
719  fS = _pstSrc->vRGB.fS;
720  fV = _pstSrc->vRGB.fV;
721 
722  /* Gray? */
723  if(fS == orxFLOAT_0)
724  {
725  /* Updates result */
726  orxVector_SetAll(&(pstResult->vRGB), fV);
727  }
728  else
729  {
730  orxFLOAT fFullHue, fSector, fIntermediate;
731 
732  /* Gets intermediate value */
733  fIntermediate = fV * fS;
734 
735  /* Gets full hue & sector */
736  fFullHue = orx2F(6.0f) * fH;
737  fSector = orxMath_Floor(fFullHue);
738 
739  /* Depending on sector */
740  switch(orxF2U(fSector))
741  {
742  default:
743  case 0:
744  {
745  /* Updates RGB components */
746  pstResult->vRGB.fR = fV;
747  pstResult->vRGB.fG = fV - (fIntermediate - (fIntermediate * (fFullHue - fSector)));
748  pstResult->vRGB.fB = fV - fIntermediate;
749 
750  break;
751  }
752 
753  case 1:
754  {
755  /* Updates RGB components */
756  pstResult->vRGB.fR = fV - (fIntermediate * (fFullHue - fSector));
757  pstResult->vRGB.fG = fV;
758  pstResult->vRGB.fB = fV - fIntermediate;
759 
760  break;
761  }
762 
763  case 2:
764  {
