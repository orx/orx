507 static orxCOLOR *orxFASTCALL orxColor_FromRGBToHSL(orxCOLOR *_pstDst, const orxCOLOR *_pstSrc)
508 {
509  orxCOLOR *pstResult = _pstDst;
510  orxFLOAT fMin, fMax, fDelta, fR, fG, fB;
511 
512  /* Checks */
513  orxASSERT(_pstDst != orxNULL);
514  orxASSERT(_pstSrc != orxNULL);
515 
516  /* Gets source red, blue and green components */
517  fR = _pstSrc->vRGB.fR;
518  fG = _pstSrc->vRGB.fG;
519  fB = _pstSrc->vRGB.fB;
520 
521  /* Gets min, max & delta values */
522  fMin = orxMIN(fR, orxMIN(fG, fB));
523  fMax = orxMAX(fR, orxMAX(fG, fB));
524  fDelta = fMax - fMin;
525 
526  /* Stores lightness */
527  pstResult->vHSL.fL = orx2F(0.5f) * (fMax + fMin);
528 
529  /* Gray? */
530  if(fDelta == orxFLOAT_0)
531  {
532  /* Gets hue & saturation */
533  pstResult->vHSL.fH = pstResult->vHSL.fS = orxFLOAT_0;
534  }
535  else
536  {
537  /* Updates saturation */
538  pstResult->vHSL.fS = (pstResult->vHSL.fL < orx2F(0.5f))
539  ? fDelta / (fMax + fMin)
540  : fDelta / (orx2F(2.0f) - fMax - fMin);
541 
542  /* Red tone? */
543  if(fR == fMax)
544  {
545  /* Updates hue */
546  pstResult->vHSL.fH = orx2F(1.0f / 6.0f) * (fG - fB) / fDelta;
547  }
548  /* Green tone? */
549  else if(fG == fMax)
550  {
551  /* Updates hue */
552  pstResult->vHSL.fH = orx2F(1.0f / 3.0f) + (orx2F(1.0f / 6.0f) * (fB - fR) / fDelta);
553  }
554  /* Blue tone */
555  else
556  {
557  /* Updates hue */
558  pstResult->vHSL.fH = orx2F(2.0f / 3.0f) + (orx2F(1.0f / 6.0f) * (fR - fG) / fDelta);
559  }
560 
561  /* Clamps hue */
562  if(pstResult->vHSL.fH < orxFLOAT_0)
563  {
564  pstResult->vHSL.fH += orxFLOAT_1;
565  }
566  else if(pstResult->vHSL.fH > orxFLOAT_1)
567  {
568  pstResult->vHSL.fH -= orxFLOAT_1;
569  }
570  }
571 
572  /* Updates alpha */
573  pstResult->fAlpha = _pstSrc->fAlpha;
574 
575  /* Done! */
576  return pstResult;
577 }

584 static orxCOLOR *orxFASTCALL orxColor_FromHSLToRGB(orxCOLOR *_pstDst, const orxCOLOR *_pstSrc)
585 {
586  orxCOLOR *pstResult = _pstDst;
587  orxFLOAT fH, fS, fL;
588 
589 #define orxCOLOR_GET_RGB_COMPONENT(RESULT, ALT, CHROMA, HUE) \
590 do \
591 { \
592  if(HUE < orx2F(1.0f / 6.0f)) \
593  { \
594  RESULT = ALT + (orx2F(6.0f) * HUE * (CHROMA - ALT)); \
595  } \
596  else if(HUE < orx2F(1.0f / 2.0f)) \
597  { \
598  RESULT = CHROMA; \
599  } \
600  else if(HUE < orx2F(2.0f / 3.0f)) \
601  { \
602  RESULT = ALT + (orx2F(6.0f) * (CHROMA - ALT) * (orx2F(2.0f / 3.0f) - HUE)); \
603  } \
604  else \
605  { \
606  RESULT = ALT; \
607  } \
608  if(RESULT < orxMATH_KF_EPSILON) \
609  { \
610  RESULT = orxFLOAT_0; \
611  } \
612  else if(RESULT > orxFLOAT_1 - orxMATH_KF_EPSILON) \
613  { \
614  RESULT = orxFLOAT_1; \
615  } \
616 } \
617 while(orxFALSE)
618 
619  /* Checks */
620  orxASSERT(_pstDst != orxNULL);
621  orxASSERT(_pstSrc != orxNULL);
622 
623  /* Gets source hue, saturation and lightness components */
624  fH = _pstSrc->vRGB.fH;
625  fS = _pstSrc->vRGB.fS;
626  fL = _pstSrc->vRGB.fL;
627 
628  /* Gray? */
629  if(fS == orxFLOAT_0)
630  {
631  /* Updates result */
632  orxVector_SetAll(&(pstResult->vRGB), fL);
633  }
634  else
635  {
636  orxFLOAT fChroma, fIntermediate;
637 
638  /* Gets chroma */
639  fChroma = (fL < orx2F(0.5f))
640  ? fL + (fL * fS)
641  : fL + fS - (fL * fS);
642 
643  /* Gets intermediate value */
644  fIntermediate = (orx2F(2.0f) * fL) - fChroma;
645 
646  /* Gets RGB components */
647  if(fH > orx2F(2.0f / 3.0f))
648  {
649  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fR, fIntermediate, fChroma, (fH - orx2F(2.0f / 3.0f)));
650  }
651  else
652  {
653  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fR, fIntermediate, fChroma, (fH + orx2F(1.0f / 3.0f)));
654  }
655  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fG, fIntermediate, fChroma, fH);
656  if(fH < orx2F(1.0f / 3.0f))
657  {
658  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fB, fIntermediate, fChroma, (fH + orx2F(2.0f / 3.0f)));
659  }
660  else
661  {
662  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fB, fIntermediate, fChroma, (fH - orx2F(1.0f / 3.0f)));
663  }
664  }
665 
666  /* Updates alpha */
667  pstResult->fAlpha = _pstSrc->fAlpha;
668 
669  /* Done! */
670  return pstResult;
671 }
