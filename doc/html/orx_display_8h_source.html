465  orxFLOAT fAlpha;

468  orxASSERT(_pstColor != orxNULL);
469

509 static orxCOLOR *orxFASTCALL orxColor_FromRGBToHSL(orxCOLOR *_pstDst, const orxCOLOR *_pstSrc)
510 {
511  orxCOLOR *pstResult = _pstDst;
512  orxFLOAT fMin, fMax, fDelta, fR, fG, fB;
513 
514  /* Checks */
515  orxASSERT(_pstDst != orxNULL);
516  orxASSERT(_pstSrc != orxNULL);
517 
518  /* Gets source red, blue and green components */
519  fR = _pstSrc->vRGB.fR;
520  fG = _pstSrc->vRGB.fG;
521  fB = _pstSrc->vRGB.fB;
522 
523  /* Gets min, max & delta values */
524  fMin = orxMIN(fR, orxMIN(fG, fB));
525  fMax = orxMAX(fR, orxMAX(fG, fB));
526  fDelta = fMax - fMin;
527 
528  /* Stores lightness */
529  pstResult->vHSL.fL = orx2F(0.5f) * (fMax + fMin);
530 
531  /* Gray? */
532  if(fDelta == orxFLOAT_0)
533  {
534  /* Gets hue & saturation */
535  pstResult->vHSL.fH = pstResult->vHSL.fS = orxFLOAT_0;
536  }
537  else
538  {
539  /* Updates saturation */
540  pstResult->vHSL.fS = (pstResult->vHSL.fL < orx2F(0.5f))
541  ? fDelta / (fMax + fMin)
542  : fDelta / (orx2F(2.0f) - fMax - fMin);
543 
544  /* Red tone? */
545  if(fR == fMax)
546  {
547  /* Updates hue */
548  pstResult->vHSL.fH = orx2F(1.0f / 6.0f) * (fG - fB) / fDelta;
549  }
550  /* Green tone? */
551  else if(fG == fMax)
552  {
553  /* Updates hue */
554  pstResult->vHSL.fH = orx2F(1.0f / 3.0f) + (orx2F(1.0f / 6.0f) * (fB - fR) / fDelta);
555  }
556  /* Blue tone */
557  else
558  {
559  /* Updates hue */
560  pstResult->vHSL.fH = orx2F(2.0f / 3.0f) + (orx2F(1.0f / 6.0f) * (fR - fG) / fDelta);
561  }
562 
563  /* Clamps hue */
564  if(pstResult->vHSL.fH < orxFLOAT_0)
565  {
566  pstResult->vHSL.fH += orxFLOAT_1;
567  }
568  else if(pstResult->vHSL.fH > orxFLOAT_1)
569  {
570  pstResult->vHSL.fH -= orxFLOAT_1;
571  }
572  }
573 
574  /* Updates alpha */
575  pstResult->fAlpha = _pstSrc->fAlpha;
576 
577  /* Done! */
578  return pstResult;
579 }

586 static orxCOLOR *orxFASTCALL orxColor_FromHSLToRGB(orxCOLOR *_pstDst, const orxCOLOR *_pstSrc)
587 {
588  orxCOLOR *pstResult = _pstDst;
589  orxFLOAT fH, fS, fL;
590 
591 #define orxCOLOR_GET_RGB_COMPONENT(RESULT, ALT, CHROMA, HUE) \
592 do \
593 { \
594  if(HUE < orx2F(1.0f / 6.0f)) \
595  { \
596  RESULT = ALT + (orx2F(6.0f) * HUE * (CHROMA - ALT)); \
597  } \
598  else if(HUE < orx2F(1.0f / 2.0f)) \
599  { \
600  RESULT = CHROMA; \
601  } \
602  else if(HUE < orx2F(2.0f / 3.0f)) \
603  { \
604  RESULT = ALT + (orx2F(6.0f) * (CHROMA - ALT) * (orx2F(2.0f / 3.0f) - HUE)); \
605  } \
606  else \
607  { \
608  RESULT = ALT; \
609  } \
610  if(RESULT < orxMATH_KF_EPSILON) \
611  { \
612  RESULT = orxFLOAT_0; \
613  } \
614  else if(RESULT > orxFLOAT_1 - orxMATH_KF_EPSILON) \
615  { \
616  RESULT = orxFLOAT_1; \
617  } \
618 } \
619 while(orxFALSE)
620 
621  /* Checks */
622  orxASSERT(_pstDst != orxNULL);
623  orxASSERT(_pstSrc != orxNULL);
624 
625  /* Gets source hue, saturation and lightness components */
626  fH = _pstSrc->vRGB.fH;
627  fS = _pstSrc->vRGB.fS;
628  fL = _pstSrc->vRGB.fL;
629 
630  /* Gray? */
631  if(fS == orxFLOAT_0)
632  {
633  /* Updates result */
634  orxVector_SetAll(&(pstResult->vRGB), fL);
635  }
636  else
637  {
638  orxFLOAT fChroma, fIntermediate;
639 
640  /* Gets chroma */
641  fChroma = (fL < orx2F(0.5f))
642  ? fL + (fL * fS)
643  : fL + fS - (fL * fS);
644 
645  /* Gets intermediate value */
646  fIntermediate = (orx2F(2.0f) * fL) - fChroma;
647 
648  /* Gets RGB components */
649  if(fH > orx2F(2.0f / 3.0f))
650  {
651  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fR, fIntermediate, fChroma, (fH - orx2F(2.0f / 3.0f)));
652  }
653  else
654  {
655  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fR, fIntermediate, fChroma, (fH + orx2F(1.0f / 3.0f)));
656  }
657  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fG, fIntermediate, fChroma, fH);
658  if(fH < orx2F(1.0f / 3.0f))
659  {
660  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fB, fIntermediate, fChroma, (fH + orx2F(2.0f / 3.0f)));
661  }
662  else
663  {
664  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fB, fIntermediate, fChroma, (fH - orx2F(1.0f / 3.0f)));
665  }
666  }
667 
668  /* Updates alpha */
669  pstResult->fAlpha = _pstSrc->fAlpha;
670 
671  /* Done! */
672  return pstResult;
673 }
