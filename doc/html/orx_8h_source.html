193 static orxINLINE void orx_Execute(orxU32 _u32NbParams, orxSTRING _azParams[], const orxMODULE_INIT_FUNCTION _pfnInit, const orxMODULE_RUN_FUNCTION _pfnRun, const orxMODULE_EXIT_FUNCTION _pfnExit)
194 {
195  /* Inits the Debug System */
196  orxDEBUG_INIT();
197 
198  /* Checks */
199  orxASSERT(_pfnRun != orxNULL);
200 
201  /* Registers main module */
202  orxModule_Register(orxMODULE_ID_MAIN, "MAIN", orx_MainSetup, _pfnInit, _pfnExit);
203 
204  /* Sends the command line arguments to orxParam module */
205  if(orxParam_SetArgs(_u32NbParams, _azParams) != orxSTATUS_FAILURE)
206  {
207  /* Sets thread callbacks */
208  orxThread_SetCallbacks(orxAndroid_JNI_SetupThread, orxNULL, orxNULL);
209 
210  /* Inits the engine */
211  if(orxModule_Init(orxMODULE_ID_MAIN) != orxSTATUS_FAILURE)
212  {
213  orxSYSTEM_EVENT_PAYLOAD stPayload;
214  orxSTATUS eClockStatus, eMainStatus;
215  orxBOOL bStop;
216 
217  /* Registers default event handler */
218  orxEvent_AddHandler(orxEVENT_TYPE_SYSTEM, orx_DefaultEventHandler);
219  orxEvent_SetHandlerIDFlags(orx_DefaultEventHandler, orxEVENT_TYPE_SYSTEM, orxNULL, orxEVENT_GET_FLAG(orxSYSTEM_EVENT_CLOSE), orxEVENT_KU32_MASK_ID_ALL);
220 
221  /* Clears payload */
222  orxMemory_Zero(&stPayload, sizeof(orxSYSTEM_EVENT_PAYLOAD));
223 
224  /* Main loop */
225  for(bStop = orxFALSE, sbStopByEvent = orxFALSE;
226  bStop == orxFALSE;
227  bStop = ((sbStopByEvent != orxFALSE) || (eMainStatus == orxSTATUS_FAILURE) || (eClockStatus == orxSTATUS_FAILURE)) ? orxTRUE : orxFALSE)
228  {
229  /* Sends frame start event */
230  orxEVENT_SEND(orxEVENT_TYPE_SYSTEM, orxSYSTEM_EVENT_GAME_LOOP_START, orxNULL, orxNULL, &stPayload);
231 
232  /* Runs game specific code */
233  eMainStatus = _pfnRun();
234 
235  /* Updates clock system */
236  eClockStatus = orxClock_Update();
237 
238  /* Sends frame stop event */
239  orxEVENT_SEND(orxEVENT_TYPE_SYSTEM, orxSYSTEM_EVENT_GAME_LOOP_STOP, orxNULL, orxNULL, &stPayload);
240 
241  /* Updates frame count */
242  stPayload.u32FrameCount++;
243  }
244 
245  /* Removes event handler */
246  orxEvent_RemoveHandler(orxEVENT_TYPE_SYSTEM, orx_DefaultEventHandler);
247

249  orxModule_Exit(orxMODULE_ID_MAIN);
250  }
251 
252  /* Clears params */

254  }
255

257  orxDEBUG_EXIT();

280  orxEVENT_SEND(orxEVENT_TYPE_SYSTEM, orxSYSTEM_EVENT_GAME_LOOP_STOP, orxNULL, orxNULL, &sstPayload);
281 
282  /* Updates frame count */
283  sstPayload.u32FrameCount++;
284 
285  /* Should stop? */
286  if((sbStopByEvent != orxFALSE) || (eMainStatus == orxSTATUS_FAILURE) || (eClockStatus == orxSTATUS_FAILURE))
287  {

290

293 
294  /* Clears params */
295  orxParam_SetArgs(0, orxNULL);
296 
297  /* Exits from the Debug system */
298  orxDEBUG_EXIT();
299 
300  /* Cancels main loop */
301  emscripten_cancel_main_loop();
302  }
