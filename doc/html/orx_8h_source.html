183 static orxINLINE void orx_Execute(orxU32 _u32NbParams, orxSTRING _azParams[], const orxMODULE_INIT_FUNCTION _pfnInit, const orxMODULE_RUN_FUNCTION _pfnRun, const orxMODULE_EXIT_FUNCTION _pfnExit)
184 {
185  /* Inits the Debug System */
186  orxDEBUG_INIT();
187 
188  /* Checks */
189  orxASSERT(_pfnRun != orxNULL);
190 
191  /* Registers main module */
192  orxModule_Register(orxMODULE_ID_MAIN, "MAIN", orx_MainSetup, _pfnInit, _pfnExit);
193 
194  /* Sends the command line arguments to orxParam module */
195  if(orxParam_SetArgs(_u32NbParams, _azParams) != orxSTATUS_FAILURE)
196  {
197  /* Sets thread callbacks */
198  orxThread_SetCallbacks(orxAndroid_JNI_SetupThread, orxNULL, orxNULL);
199 
200  /* Inits the engine */
201  if(orxModule_Init(orxMODULE_ID_MAIN) != orxSTATUS_FAILURE)
202  {
203  orxSYSTEM_EVENT_PAYLOAD stPayload;
204  orxSTATUS eClockStatus, eMainStatus;
205  orxBOOL bStop;
206 
207  /* Registers default event handler */
208  orxEvent_AddHandler(orxEVENT_TYPE_SYSTEM, orx_DefaultEventHandler);
209  orxEvent_SetHandlerIDFlags(orx_DefaultEventHandler, orxEVENT_TYPE_SYSTEM, orxNULL, orxEVENT_GET_FLAG(orxSYSTEM_EVENT_CLOSE), orxEVENT_KU32_MASK_ID_ALL);
210 
211  /* Clears payload */
212  orxMemory_Zero(&stPayload, sizeof(orxSYSTEM_EVENT_PAYLOAD));
213 
214  /* Main loop */
215  for(bStop = orxFALSE, sbStopByEvent = orxFALSE;
216  bStop == orxFALSE;
217  bStop = ((sbStopByEvent != orxFALSE) || (eMainStatus == orxSTATUS_FAILURE) || (eClockStatus == orxSTATUS_FAILURE)) ? orxTRUE : orxFALSE)
218  {
219  orxAndroid_PumpEvents();
220 
221  /* Sends frame start event */
222  orxEVENT_SEND(orxEVENT_TYPE_SYSTEM, orxSYSTEM_EVENT_GAME_LOOP_START, orxNULL, orxNULL, &stPayload);
223 
224  /* Runs game specific code */
225  eMainStatus = _pfnRun();
226 
227  /* Updates clock system */
228  eClockStatus = orxClock_Update();
229 
230  /* Sends frame stop event */
231  orxEVENT_SEND(orxEVENT_TYPE_SYSTEM, orxSYSTEM_EVENT_GAME_LOOP_STOP, orxNULL, orxNULL, &stPayload);
232 
233  /* Updates frame count */
234  stPayload.u32FrameCount++;
235  }
236 
237  /* Removes event handler */
238  orxEvent_RemoveHandler(orxEVENT_TYPE_SYSTEM, orx_DefaultEventHandler);
239 
240  /* Exits from the engine */
241  orxModule_Exit(orxMODULE_ID_MAIN);
242  }
243  }
244 
245  /* Exits from the Debug system */
246  orxDEBUG_EXIT();
247 }
