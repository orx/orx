401 static orxINLINE orxSTATUS orx_LogAllStructures()
402 {
403 #define orx_MAX_NAME_LENGTH 20
404 #define orx_DECLARE_STRUCTURE_INFO(ID, FN) {orxSTRUCTURE_ID_##ID, (orxSTRUCTURE_GETNAME)FN}

406  typedef orxSTRING(orxFASTCALL *orxSTRUCTURE_GETNAME)(const void *_pStructure);
407 
408  static struct
409  {
410  orxSTRUCTURE_ID eID;
411  orxSTRUCTURE_GETNAME pfnGetName;
412  }
413  sastStructureInfoList[] =
414  {
415  orx_DECLARE_STRUCTURE_INFO(OBJECT, orxObject_GetName),
416  orx_DECLARE_STRUCTURE_INFO(FONT, orxFont_GetName),
417  orx_DECLARE_STRUCTURE_INFO(VIEWPORT, orxViewport_GetName),
418  orx_DECLARE_STRUCTURE_INFO(ANIMSET, orxAnimSet_GetName),
419  orx_DECLARE_STRUCTURE_INFO(ANIM, orxAnim_GetName),
420  orx_DECLARE_STRUCTURE_INFO(BODY, orxBody_GetName),
421  orx_DECLARE_STRUCTURE_INFO(CAMERA, orxCamera_GetName),
422  orx_DECLARE_STRUCTURE_INFO(CLOCK, orxClock_GetName),
423  orx_DECLARE_STRUCTURE_INFO(FX, orxFX_GetName),
424  orx_DECLARE_STRUCTURE_INFO(GRAPHIC, orxGraphic_GetName),
425  orx_DECLARE_STRUCTURE_INFO(SHADER, orxShader_GetName),
426  orx_DECLARE_STRUCTURE_INFO(SOUND, orxSound_GetName),
427  orx_DECLARE_STRUCTURE_INFO(SPAWNER, orxSpawner_GetName),
428  orx_DECLARE_STRUCTURE_INFO(TEXT, orxText_GetName),
429  orx_DECLARE_STRUCTURE_INFO(TEXTURE, orxTexture_GetName),
430  orx_DECLARE_STRUCTURE_INFO(TIMELINE, orxNULL)
431  };
432  orxU32 i;
433  orxHASHTABLE *pstTable;
434  orxSTATUS eResult = orxSTATUS_FAILURE;
435 
436  /* Creates table */
437  pstTable = orxHashTable_Create(1024, orxHASHTABLE_KU32_FLAG_NONE, orxMEMORY_TYPE_TEMP);
438 
439  /* Valid? */
440  if(pstTable != orxNULL)
441  {
442  orxCHAR acBuffer[1024] = {0};
443  orxU32 u32DebugFlags;
444 
445  /* Backups debug flags */
446  u32DebugFlags = orxDEBUG_GET_FLAGS();
447 
448  /* Sets new debug flags */
449  orxDEBUG_SET_FLAGS(orxDEBUG_KU32_STATIC_FLAG_NONE,
450  orxDEBUG_KU32_STATIC_FLAG_TIMESTAMP | orxDEBUG_KU32_STATIC_FLAG_TYPE);
451 
452  /* Logs header */
453  orxLOG("*** BEGIN STRUCTURE LOG ***\n");
454 
455  /* For all IDs */
456  for(i = 0; i < orxARRAY_GET_ITEM_COUNT(sastStructureInfoList); i++)
457  {
458  orxSTRUCTURE *pstStructure;
459 
460  /* Checks */
461  orxASSERT(orxStructure_GetStorageType(sastStructureInfoList[i].eID) == orxSTRUCTURE_STORAGE_TYPE_LINKLIST);
462 
463  /* For all structures */
464  for(pstStructure = orxStructure_GetFirst(sastStructureInfoList[i].eID);
465  pstStructure != orxNULL;
466  pstStructure = orxStructure_GetNext(pstStructure))
467  {
468  /* Is owner-less? */
469  if(orxStructure_GetOwner(pstStructure) == orxNULL)
470  {
471  /* Not already logged? */
472  if(orxHashTable_Get(pstTable, orxStructure_GetGUID(pstStructure)) == orxNULL)
473  {
474  /* Adds it to table */
475  orxHashTable_Set(pstTable, orxStructure_GetGUID(pstStructure), pstStructure);
476 
477  /* Logs it */
478  if(sastStructureInfoList[i].pfnGetName != orxNULL)
479  {
480  orxU32 u32Offset;
481  u32Offset = orxString_NPrint(acBuffer, sizeof(acBuffer) - 1, "%-8s \"%s\"", orxStructure_GetIDString(sastStructureInfoList[i].eID), sastStructureInfoList[i].pfnGetName(pstStructure));
482  orxLOG("%s%*s[% 016llX]\n", acBuffer, orxMAX(0, orx_MAX_NAME_LENGTH + 12 - u32Offset), orxSTRING_EMPTY, pstStructure->u64GUID);
483  }
484  else
485  {
486  orxLOG("%-8s %*s [%016llX]\n", orxStructure_GetIDString(sastStructureInfoList[i].eID), orx_MAX_NAME_LENGTH + 2, orxSTRING_EMPTY, pstStructure->u64GUID);
487  }
488 
490  }
491  }
492  }
493  }
494 
495  /* Logs footer */
496  orxLOG("*** END STRUCTURE LOG ***");
497 
498  /* Restores debug flags */
499  orxDEBUG_SET_FLAGS(u32DebugFlags, orxDEBUG_KU32_STATIC_MASK_USER_ALL);
500 
501  /* Updates result */
502  eResult = orxSTATUS_SUCCESS;
503 
504  /* Deletes table */
505  orxHashTable_Delete(pstTable);
506  }
507

509 #undef orx_MAX_NAME_LENGTH
510 
511  /* Done! */
512  return eResult;
513 }
